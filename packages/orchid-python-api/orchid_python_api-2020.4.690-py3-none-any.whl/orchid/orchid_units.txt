# Ratio units

# Density
pound_per_cubic_foot = lb / cu_ft = lb_per_cu_ft
kilogram_per_cubic_meter = kg / (m ** 3) = kg_per_cu_m

# Proppant concentration
[proppant_concentration] = [mass] / [volume]
pound_per_gallon = lb / gal = lb_per_gal

# Slurry rate
[slurry_rate] = [mass] / [time]
oil_barrel_per_minute = oil_bbl / min = bpm
cubic_meter_per_minute = (m ** 3) / min = cu_m_per_min

# Volume
cubic_meter = (m ** 3) = cu_m
